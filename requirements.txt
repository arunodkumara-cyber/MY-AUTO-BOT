# Script to generate a clean requirements.txt for KD AI AUTO BOT
libraries = [
    "streamlit",
    "pandas",
    "numpy",
    "plotly",
    "requests"
]

def create_requirements():
    try:
        # Writing in 'w' mode to overwrite any previous broken file
        with open("requirements.txt", "w", encoding="utf-8") as f:
            for lib in libraries:
                f.write(f"{lib}\n")
        print("✅ Success: 'requirements.txt' created perfectly.")
    except Exception as e:
        print(f"❌ Error creating file: {e}")

if __name__ == "__main__":
    create_requirements()
